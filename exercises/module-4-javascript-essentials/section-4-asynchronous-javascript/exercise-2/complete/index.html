Async/Await Refactoring

// Original callback-based code function getMedicationSchedule(patientId, callback) { fetchPatient(patientId, (error, patient) => { if (error) { callback(error); return; } fetchMedications(patient.id, (error, medications) => { if (error) { callback(error); return; } generateSchedule(medications, (error, schedule) => { if (error) { callback(error); return; } callback(null, schedule); }); }); }); }
// Refactored async/await version async function getMedicationSchedule(patientId) { try { const patient = await new Promise((resolve, reject) => { fetchPatient(patientId, (error, result) => { if (error) reject(error); else resolve(result); }); }); const medications = await new Promise((resolve, reject) => { fetchMedications(patient.id, (error, result) => { if (error) reject(error); else resolve(result); }); }); const schedule = await new Promise((resolve, reject) => { generateSchedule(medications, (error, result) => { if (error) reject(error); else resolve(result); }); }); return schedule; } catch (error) { throw error; } }
